**In app/src/services/youtube.service.ts:**

Add a `searchShorts` function that searches YouTube for short-form videos. Follow the exact pattern of the existing `searchVideos` function but with these modifications:

1. Append ` #Shorts` to the search query to bias toward short-form content
2. Add `&videoDuration=short` parameter to the API URL (filters to videos under 4 minutes)
3. Set `maxResults` to 2 (we want fewer shorts than regular videos per batch)

```typescript
async function searchShorts(
  query: string,
  apiKey: string,
  maxResults = 2,
): Promise<Array<{ videoId: string; title: string; channelTitle: string; thumbnailUrl: string }>> {
  const shortsQuery = `${query} #Shorts`;
  // Use the same YOUTUBE_SEARCH_URL as searchVideos, adding videoDuration=short
  const url = `${YOUTUBE_SEARCH_URL}?part=snippet&type=video&videoEmbeddable=true` +
    `&videoDuration=short&q=${encodeURIComponent(shortsQuery)}&maxResults=${maxResults}` +
    `&relevanceLanguage=en&safeSearch=strict&key=${apiKey}`;

  const res = await fetch(url);
  if (!res.ok) return [];

  const data = await res.json();
  return (data.items ?? []).map((item: any) => ({
    videoId: item.id?.videoId ?? '',
    title: item.snippet?.title ?? '',
    channelTitle: item.snippet?.channelTitle ?? '',
    // Use maxresdefault for potentially portrait thumbnails; fallback to high
    thumbnailUrl: `https://img.youtube.com/vi/${item.id?.videoId}/maxresdefault.jpg`,
  })).filter((v: any) => v.videoId);
}
```

Add a `generateShortPosts` function that creates `DailyPost` objects with `sourceType: 'short'`:

```typescript
async function generateShortPosts(
  questions: Question[],
  count = 2,
): Promise<DailyPost[]> {
  const settings = settingsService.getSync();
  const apiKey = settings.youtube.apiKey;
  if (!apiKey) return [];

  // Pick random questions to search for shorts
  const shuffled = [...questions].sort(() => Math.random() - 0.5);
  const searchQuestions = shuffled.slice(0, count);

  const posts: DailyPost[] = [];
  for (const q of searchQuestions) {
    const searchTerm = q.title || q.content.slice(0, 60);
    const results = await searchShorts(searchTerm, apiKey, 1);
    if (results.length === 0) continue;

    const video = results[0];
    const post: DailyPost = {
      id: `short-${video.videoId}`,
      date: today(),
      title: video.title,
      teaser: {
        hook: q.title || q.content.slice(0, 80),
        preview: '',
      },
      bodyMarkdown: '',
      whyCare: '',
      takeaway: '',
      quickAskPrompts: [],
      narrativeMode: 'explain' as PostNarrativeMode,
      contextLabel: video.channelTitle,
      sourceType: 'short',
      sourceQuestionIds: [q.id],
      sourceQuestionTitles: [q.title || q.content.slice(0, 50)],
      keywords: q.keywords.slice(0, 3),
      generatedAt: Date.now(),
      origin: 'ai',
      presentationStyle: 'short',
      videoMeta: {
        videoId: video.videoId,
        channelTitle: video.channelTitle,
        thumbnailUrl: video.thumbnailUrl,
        summary: q.shortSummary || q.summary?.slice(0, 200) || '',
      },
    };
    posts.push(post);
  }
  return posts;
}
```

Also add a `getCachedShortPosts` function that reads from a separate localStorage key `echolearn_short_posts`:
```typescript
const SHORTS_STORAGE_KEY = 'echolearn_short_posts';

function getCachedShortPosts(): DailyPost[] {
  try {
    const raw = localStorage.getItem(SHORTS_STORAGE_KEY);
    if (!raw) return [];
    const data = JSON.parse(raw) as { date: string; posts: DailyPost[] };
    if (data.date !== today()) return [];
    return data.posts;
  } catch { return []; }
}

function cacheShortPosts(posts: DailyPost[]): void {
  localStorage.setItem(SHORTS_STORAGE_KEY, JSON.stringify({ date: today(), posts }));
}
```

Call `cacheShortPosts(posts)` at the end of `generateShortPosts` before returning.
