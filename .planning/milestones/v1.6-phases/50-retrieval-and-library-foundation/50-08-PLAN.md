export function CollectionDrillInScreen() {
  const { id } = useParams<{ id: string }>();
  const navigate = useNavigate();
  const { t } = useTranslation();

  const [collection, setCollection] = useState(() => id ? collectionService.getCollections().find(c => c.id === id) : undefined);
  const [posts, setPosts] = useState(() => id ? collectionService.getCollectionPosts(id) : []);
  const [kebabOpen, setKebabOpen] = useState(false);
  const [renameOpen, setRenameOpen] = useState(false);
  const [deleteOpen, setDeleteOpen] = useState(false);
  const [renameValue, setRenameValue] = useState(collection?.name ?? '');
  const [renameError, setRenameError] = useState<string | null>(null);
  // LongPressMenu host state — drill-in owns its own menu:
  const [menuOpen, setMenuOpen] = useState(false);
  const [menuPostId, setMenuPostId] = useState<string | null>(null);
  const [menuAnchorId, setMenuAnchorId] = useState<string | null>(null);
  // Optional: CollectionPickerSheet host state (so user can save a post to OTHER collections from drill-in):
  const [pickerOpen, setPickerOpen] = useState(false);
  const [pickerPostId, setPickerPostId] = useState<string | null>(null);

  // COLLECTIONS_CHANGED subscription — refresh in place; if collection deleted, navigate back:
  useEffect(() => {
    const unsub = eventBus.subscribe('COLLECTIONS_CHANGED', (e) => {
      if (!id) return;
      const updated = collectionService.getCollections().find(c => c.id === id);
      if (!updated) { navigate('/saved'); return; }
      setCollection(updated);
      setPosts(collectionService.getCollectionPosts(id));
    });
    return unsub;
  }, [id, navigate]);

  // Render: Header (portaled) + scroll container with SavedRow-equivalent rows + LongPressMenu with collectionContext.
  // Not-found guard early-return matches AnchorDetailScreen.tsx pattern.
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user input → collection name → header title | Header renders `collection.name` directly as JSX text. React-escaped. |
| Delete confirmation → service write | Two-step (kebab → confirm sheet → Delete button) prevents accidental data loss. |
| Orphan postIds in collection.postIds | Posts purged from history must not throw at render. |
| Header DOM positioning | Phase 32.1 portal rule — no transform/will-change/filter/contain on ancestors. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-50-XSS-NAME | Tampering / XSS | Collection name in Header title + Rename/Delete sheet copy + toast | mitigate | React text-node escaping. Source test asserts no `dangerouslySetInnerHTML`. Validation already enforced in 50-03 (trim, ≤50 chars, dedup). |
| T-50-ORPHAN | Spoofing / Stale Data | post.id present in collection.postIds but absent from postHistoryService | mitigate | getCollectionPosts() (from 50-03) silently drops orphans. Render branch shows empty-state when posts.length === 0 even if collection.postIds.length > 0. |
| T-50-HEADER-PORTAL | Availability / UX | Drill-in Header positioning | mitigate | CLAUDE.md §"Header positioning" rule 1: outer container uses `display: flex; flexDirection: column; minHeight: '100%'` ONLY — no transform / will-change / filter / contain / perspective. Source test asserts. |
| T-50-DOUBLE-DELETE | Race / Data Loss | Rapid double-tap on Delete | accept | collectionService.deleteCollection is idempotent on missing id (second call no-ops). Worst case: two toasts fire. Acceptable risk. |
</threat_model>
