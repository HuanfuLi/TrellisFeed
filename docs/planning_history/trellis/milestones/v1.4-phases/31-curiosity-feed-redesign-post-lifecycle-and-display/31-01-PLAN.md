    **Step 2 — Create app/src/services/post-queue.service.ts:**

    localStorage key: `echolearn_post_queue`
    Stored shape: `{ date: string, posts: DailyPost[], cycleNumber: number }`

    ```typescript
    import type { DailyPost } from '../types';

    const STORAGE_KEY = 'echolearn_post_queue';
    const REFILL_THRESHOLD = 8;

    interface QueueState {
      date: string;
      posts: DailyPost[];
      cycleNumber: number;
    }

    function today(): string {
      return new Date().toISOString().slice(0, 10);
    }

    function load(): QueueState {
      try {
        const raw = localStorage.getItem(STORAGE_KEY);
        if (!raw) return { date: today(), posts: [], cycleNumber: 0 };
        const parsed = JSON.parse(raw) as QueueState;
        if (parsed.date !== today()) {
          // Date mismatch — return empty for today (warm-start handled by caller)
          return { date: today(), posts: [], cycleNumber: 0 };
        }
        return parsed;
      } catch {
        return { date: today(), posts: [], cycleNumber: 0 };
      }
    }

    function save(state: QueueState): void {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
    }

    let _state: QueueState = load();

    export const postQueueService = {
      getQueue(): DailyPost[] { return [..._state.posts]; },
      enqueue(posts: DailyPost[]): void {
        _state.posts.push(...posts);
        save(_state);
      },
      dequeue(count: number): DailyPost[] {
        const items = _state.posts.splice(0, count);
        save(_state);
        return items;
      },
      size(): number { return _state.posts.length; },
      needsRefill(): boolean { return _state.posts.length < REFILL_THRESHOLD; },
      getCycleNumber(): number { return _state.cycleNumber; },
      incrementCycle(): void {
        _state.cycleNumber++;
        save(_state);
      },
      resetForNewDay(): void {
        _state = { date: today(), posts: [], cycleNumber: 0 };
        save(_state);
      },
      loadQueue(): void {
        _state = load();
      },
      getYesterdayQueue(): DailyPost[] {
        try {
          const raw = localStorage.getItem(STORAGE_KEY);
          if (!raw) return [];
          const parsed = JSON.parse(raw) as QueueState;
          if (parsed.date === today()) return []; // not yesterday
          return parsed.posts || [];
        } catch { return []; }
      },
    };
    ```

    ```typescript
    import type { DailyPost } from '../types';
    import { settingsService } from './mock/settings.mock';

    const STORAGE_KEY = 'echolearn_post_history';
    const DEFAULT_RETENTION_DAYS = 7;

    function loadPosts(): DailyPost[] {
      try {
        const raw = localStorage.getItem(STORAGE_KEY);
        if (!raw) return [];
        return JSON.parse(raw) as DailyPost[];
      } catch { return []; }
    }

    function savePosts(posts: DailyPost[]): void {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(posts));
    }

    export const postHistoryService = {
      addPost(post: DailyPost): void {
        const posts = loadPosts();
        if (posts.some(p => p.id === post.id)) return; // dedup
        posts.push(post);
        savePosts(posts);
      },

      getPosts(): DailyPost[] {
        return loadPosts().sort((a, b) => b.generatedAt - a.generatedAt);
      },

      getPostsByDay(): Map<string, DailyPost[]> {
        const posts = this.getPosts();
        const grouped = new Map<string, DailyPost[]>();
        for (const post of posts) {
          const day = post.date;
          const arr = grouped.get(day) || [];
          arr.push(post);
          grouped.set(day, arr);
        }
        return grouped;
      },

      purgeExpired(): void {
        const settings = settingsService.getSync();
        const retentionDays = settings.feed?.postRetentionDays;
        if (retentionDays === null || retentionDays === undefined) return; // keep all
        const cutoff = Date.now() - retentionDays * 24 * 60 * 60 * 1000;
        const posts = loadPosts().filter(p => p.generatedAt > cutoff);
        savePosts(posts);
      },

      clear(): void {
        localStorage.removeItem(STORAGE_KEY);
      },
    };
    ```
