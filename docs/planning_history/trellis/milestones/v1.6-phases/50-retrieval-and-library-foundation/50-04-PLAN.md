import Fuse from 'fuse.js';
import type { DailyPost } from '../types/index.ts';

// Weighted multi-field — title (0.6) > body (0.3) > source/concept (0.1)
const FUSE_OPTIONS: Fuse.IFuseOptions<DailyPost> = {
  keys: [
    { name: 'title',                 weight: 0.6 },
    { name: 'bodyMarkdown',          weight: 0.3 },
    { name: 'sourceQuestionTitles',  weight: 0.1 },
    { name: 'contextLabel',          weight: 0.1 },
  ],
  threshold: 0.35,
  ignoreLocation: true,    // CRITICAL — Pitfall 1; body matches past position ~60 silently dropped without this
  includeMatches: true,    // returns indices for HighlightedText rendering in 50-06
  includeScore: true,      // enables relevance sort (D-14)
  minMatchCharLength: 2,
  shouldSort: true,
};

// Public API:
//   buildIndex(posts: DailyPost[]): Fuse<DailyPost>            // consumers wrap in useMemo
//   search(index, query): Array<{ item: DailyPost; score?: number; matches?: Fuse.FuseResultMatch[] }>
//   capQuery(query: string): string                              // 200-char hard cap

//   dateFilter(generatedAt: number, filter: 'today'|'last7'|'last30'|'all'): boolean

<!-- Date filter math (from RESEARCH Pattern 8 — pure Date arithmetic, no lib/date.ts import to keep leaf-discipline) -->

function dateFilter(generatedAt: number, filter: 'today'|'last7'|'last30'|'all'): boolean {
  if (filter === 'all') return true;
  const now = Date.now();
  const dayMs = 24 * 60 * 60 * 1000;
  if (filter === 'today') {
    const todayStart = new Date();
    todayStart.setHours(0, 0, 0, 0);
    return generatedAt >= todayStart.getTime();
  }
  if (filter === 'last7') return generatedAt >= now - 7 * dayMs;
  if (filter === 'last30') return generatedAt >= now - 30 * dayMs;
  return true;
}

<!-- Snippet extraction (from RESEARCH Pattern 4 + UI-SPEC Surface 7) -->

function extractSnippet(body: string, firstMatchStart: number, maxChars = 120): { text: string; offset: number } {
  const half = Math.floor(maxChars / 2);
  const rawStart = Math.max(0, firstMatchStart - half);
  const rawEnd = Math.min(body.length, rawStart + maxChars);
  const text = body.slice(rawStart, rawEnd);
  const prefix = rawStart > 0 ? '…' : '';
  const suffix = rawEnd < body.length ? '…' : '';
  return { text: prefix + text + suffix, offset: rawStart - prefix.length };  // offset is the index of text[0] in body
}
