```typescript
import { questionService } from '../services/question.service';

export function resolveAnchorId(qaId: string): string | null {
  const qa = questionService.getById(qaId);
  if (!qa) return null;
  if (qa.isAnchorNode) return qa.id;
  if (!qa.parentId) return null;
  const parent = questionService.getById(qa.parentId);
  return parent?.isAnchorNode ? parent.id : null;
}
```

```typescript
function getBadgeLeafState(qaId: string): 'attention' | 'normal' {
  const anchorId = resolveAnchorId(qaId);
  if (!anchorId) return 'normal';
  const anchor = questionService.getById(anchorId);
  if (!anchor) return 'normal';
  const qaChildren = questionService.getAll().filter(q => q.parentId === anchorId && !q.isAnchorNode);
  const ls = computeLeafState(anchor, qaChildren);
  return ls === 'dying' || ls === 'falling' || ls === 'dead' ? 'attention' : 'normal';
}
```

(Performance note: this runs per-badge per-render. For typical feeds of 8-32 tiles × 1-2 badges each, the localStorage reads are well under 5ms total. Don't over-optimize.)
