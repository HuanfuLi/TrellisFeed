  // Per RESEARCH Summary point 6 (inverse-verb-with-swapped-snapshots):
  // undo of a `rename` writes ANOTHER `rename` entry with swapped before/after.
  // The `cmd` literal-union is therefore the six real command verbs — NO
  // synthetic `'undo'` literal. This keeps the reorg-prompt history honest:
  // "user did X, then user did the reverse of X" reads as two ordinary commands.
  export interface GraphEditLogEntry {
    id: string;                  // `${Date.now()}-${Math.random().toString(36).slice(2,7)}`
    ts: number;                  // Date.now()
    cmd: 'rename' | 'move' | 'merge' | 'detach' | 'prune' | 'delete';
    targetIds: string[];
    before: Record<string, unknown>;
    after: Record<string, unknown>;
  }
